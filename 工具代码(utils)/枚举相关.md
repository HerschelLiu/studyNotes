/**
 * 获取枚举的不重复数组
 */
// 小程序
export function getEnumArray(object: any): any[] {
  const arr: any[] = []
  for (const key in object) {
    const isNAN = isNaN(Number(key))
    if (isNAN) {
      arr.push({
        key: isNAN ? object[key] : Number(object[key]),
        value: key
      })
    }
  }
  return arr.sort((p, n) => Number(p.key - n.key))
}

// vue+ts
export function getEnumArray < T = number >(object: object): EnumArray < T > [] {
  return Object.entries(object).filter(item => /^[A-Za-z0-9-_]+$/.test(item[1]) || item[1] === '').map(item => {
    return {
      key: item[1] === '' ? '' : isNaN(Number(item[1])) ? item[1] : Number(item[1]),
      value: item[0]
    }
  }) as any
}
export interface EnumArray < T > {
  key: T;
  value: string;
}

/**
 * 直接从枚举改造对象中取值
 */
// 小程序
export function getValueByEnum(key: string | number, obj: any[]): string {
  for (const forkey in obj) {
    if (key === obj[forkey].key) {
      return obj[forkey].value
    }
  }
  return ''
}

// vue+ts
export function getValueByEnum < T = number | string >(key: T, obj: EnumArray < T > []): string {
  for (const forkey in obj) {
    if (key === obj[forkey].key) {
      return obj[forkey].value
    }
  }
  return ''
}
