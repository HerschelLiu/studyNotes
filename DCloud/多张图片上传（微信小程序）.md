addImg() {
    uni.chooseImage({
        count: this.chooseImgCount
    }).then(data => {
        let [err, res] = data;
        if(!!err) {
            return;
        }
        let imgPathArr = res.tempFilePaths;
        uni.showLoading({
            title: '审核图片中',
        });
        Promise.all(res.tempFilePaths.map((item) => {
            return uni.uploadFile({
                url: common.apiUrl + 'api/OssUpload/uploadFileCheck',
                filePath: item,
                name: 'file',
                formData: {
                    'user': 'test'
                }
            })
        })).then(data => {
            let errImgNum = 0;
            let _imgArr = [];
            data.forEach((item, index) => {
                let [err, res] = item;
                let _result = typeof res.data == 'string' ? JSON.parse(res.data) : res.data,
                    _res = _result.data;
                if (_result.status == 200) {
                    _imgArr.push({
                        pic_id: _res.pic_id,
                        pic_path: imgPathArr[index]
                    });
                } else{
                    errImgNum++;
                }
            });
            uni.hideLoading();
            if(errImgNum > 0) {
                uni.showToast({
                    title: `${errImgNum}张违规图片`,
                    icon: 'none',
                    duration: 2000
                });
            }

            this.imgs = [...this.imgs, ..._imgArr];
        });
    });
},
uploadImg(imgPath, type, index) {
    uni.showLoading({
        title: '审核图片中',
        mask: true
    });
    uni.uploadFile({
        url: common.apiUrl + 'api/OssUpload/uploadFileCheck',
        filePath: imgPath,
        name: 'file',
        formData: {
            'user': 'test'
        }
    }).then(data => {
        let [err, res] = data;
        uni.hideLoading();
        if (!!err) {
            uni.showToast({
                title: '请求超时，请稍后重试',
                icon: 'none',
                duration: 1500
            });
            uni.hideLoading();
            return;
        }

        let _result = typeof res.data == 'string' ? JSON.parse(res.data) : res.data,
            _res = _result.data;

        if(_result.status == 200) {
            switch (type){
                case 'add':
                    this.imgs.push({
                        pic_id: _res.pic_id,
                        pic_path: imgPath
                    });
                    break;
                case 'change':
                    console.log(111);
                    this.imgs.splice(index, 1, {
                        pic_id: _res.pic_id,
                        pic_path: imgPath
                    });
                    break;
                default:
                    break;
            }
        } else {
            uni.showToast({
                title: '1张违规图片',
                icon: 'none',
                duration: 2000
            });
        }
        uni.hideLoading();
    });
},
clickImg(index) {
    this.imgs.splice(index, 1);
},

            return;
