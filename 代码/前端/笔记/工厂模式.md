```ts
enum CarType {
  Bmw = 0,
  Audit = 1,
  Benz = 2
}
```



常用创建类

```ts
class Car {
  name: string
}

class Bmw extends Car {}
class Audit extends Car {}
class Benz extends Car {}
```



工厂模式

```ts
class Car {
  name: string
  
  // 工厂方法
  static Create(carType: CarType): Car {
    let car: Car
    switch (carType) {
      case CarType.Bmw:
        car = new Bmw()
        break
      case CarType.Audio:
        car = new Audio()
        break
      case CarType.Benz:
        car = new Benz()
        break
    }
    
    return car
  }
}

class Bmw extends Car {}
class Audit extends Car {}
class Benz extends Car {}

const bmw = Car.Create(CarType.Bmw)
```
