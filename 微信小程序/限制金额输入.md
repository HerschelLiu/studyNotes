```js
/**
   * 输入
   */
  handleInput(e: WechatMiniprogram.InputEvent) {
    this.setData({
      customPrice: e.detail.value.replace(/[^\d\.]|^\./g, '').replace(/\.{2}/g, '.').replace(/^([1-9]\d*|0)(\.\d{1,2})(\.|\d{1})?$/, '$1$2').replace(/^0\d{1}/g, '0')
    })
  }

<input
  value="{{ customPrice }}"
  type="digit"
  placeholder="请输入充值金额"
  confirm-type="done"
  class="recharge-item-input {{ activeIndex === -1 ? 'active' : '' }}"
  bindfocus="handleInputFocus"
  bindinput="handleInput"
/>
```
